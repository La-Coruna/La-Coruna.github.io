using namespace std;

int N, K;
string words[50];
bool canRead[26];
int maxCnt = 0;

void func(int k, int idx) {
	if (k == K-5) {
		int cnt = 0;
		for (int i = 0; i < N; i++) {
			bool canReadWord = true;
			for (char c : words[i]) {
				if (canRead[c-'a']) continue;
				canReadWord = false;
				break;
			}
			if (canReadWord) cnt++;
		}
		maxCnt = max(maxCnt, cnt);
		if (maxCnt == N) {
			cout << maxCnt;
			exit(0);
		}
		return;
	}
	for(int i = idx; i < 26; i++){
		if (canRead[i]) continue;
		canRead[i] = true;
		func(k + 1, i);
		canRead[i] = false;
	}
}
