int N, K;
priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;
int vis[100'001];

int main(void){
    ios_base::sync_with_stdio(0);
    cin.tie(0);

    cin >> N >> K;

    vis[N] = 1;
    pq.emplace(vis[N],N);
    while(!pq.empty()){
        int cur = pq.top().Y; pq.pop();
        // cout << cur << "(" << vis[cur] << ")" << " -> ";

        if(cur == K) {
            cout << vis[cur] - 1;
            break;
        }

        for(int nxt: {cur * 2, cur - 1, cur + 1}){
            if(nxt < 0 || nxt > 100'000 || vis[nxt] != 0) continue;
            if(nxt == cur * 2){
                // ê±°ë¦¬ê°€ ë” ë©€ì–´ì§€ë©´ ì¡°ê¸° ì¢…ë£Œ.
                int cur_distance = abs(K - cur);
                int nxt_distance = abs(K - nxt);
                if(nxt_distance >= cur_distance) continue;

                vis[nxt] = vis[cur];
                pq.emplace(vis[nxt], nxt);
            } else{
                vis[nxt] = vis[cur]+1;
                pq.emplace(vis[nxt], nxt);
            }
        }
    }

    return 0;
}
