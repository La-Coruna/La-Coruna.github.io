	vector<bool>isUnselected(L, 0);
	isUnselected.resize(C, 1);

	do {
		bool hasMother = false;
		int hasSon = 0;

		// ë‹¨ì–´ ìƒì„±
		string pw = "";
		for (int i = 0; i < spell.size(); i++) {

				hasSon++;

		}

}
